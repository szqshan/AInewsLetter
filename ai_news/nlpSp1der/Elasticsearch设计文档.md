from elasticsearch import Elasticsearch, helpers

import numpy as np

es = Elasticsearch(
    ['http://localhost:9200'],
    http_auth=('elastic', 'password')

# ÂàõÂª∫Á¥¢Âºï
def create_index():
    with open('index_mapping.json', 'r') as f:
        mapping = json.load(f)
    
    if es.indices.exists(index='newsletter_articles'):
        es.indices.delete(index='newsletter_articles')
    
    es.indices.create(index='newsletter_articles', body=mapping)

# ÊâπÈáèÂØºÂÖ•Êï∞ÊçÆ
def bulk_import_articles(articles_dir):
    actions = []
    
    for article_file in Path(articles_dir).glob('*/metadata.json'):
        with open(article_file, 'r') as f:
            article = json.load(f)
        
        # ËØªÂèñMarkdownÂÜÖÂÆπ
        md_file = article_file.parent / 'content.md'
        if md_file.exists():
            with open(md_file, 'r') as f:
                article['content'] = f.read()
        
        # ÂáÜÂ§áESÊñáÊ°£
        doc = {
            '_index': 'newsletter_articles',
            '_id': article['id'],
            '_source': article
        }
        
        actions.append(doc)
        
        # ÊØè100Êù°ÊâπÈáèÂØºÂÖ•‰∏ÄÊ¨°
        if len(actions) >= 100:
            helpers.bulk(es, actions)
            actions = []
    
    # ÂØºÂÖ•Ââ©‰ΩôÊï∞ÊçÆ
    if actions:
        helpers.bulk(es, actions)

# ÁîüÊàêÂêëÈáèÔºàÁ§∫‰æãÔºåÂÆûÈôÖÈúÄË¶Å‰ΩøÁî®Ê®°ÂûãÔºâ
def generate_embedding(text):
    # ËøôÈáåÂ∫îËØ•Ë∞ÉÁî®ÂÆûÈôÖÁöÑÂêëÈáèÂåñÊ®°Âûã
    # Á§∫‰æãÔºö‰ΩøÁî®sentence-transformers
    from sentence_transformers import SentenceTransformer
    model = SentenceTransformer('all-MiniLM-L6-v2')
    embedding = model.encode(text)
    return embedding.tolist()

# Êõ¥Êñ∞ÊñáÊ°£ÂêëÈáè
def update_embeddings():
    # Ëé∑ÂèñÊâÄÊúâÊñáÊ°£
    response = es.search(
        index='newsletter_articles',
        body={'query': {'match_all': {}}, 'size': 1000}
    )
    
    for hit in response['hits']['hits']:
        doc = hit['_source']
        text = f"{doc.get('title', '')} {doc.get('subtitle', '')} {doc.get('description', '')}"
        
        embedding = generate_embedding(text)
        
        # Êõ¥Êñ∞ÊñáÊ°£
        es.update(
            index='newsletter_articles',
            id=hit['_id'],
            body={'doc': {'content_embedding': embedding}}
        )

if __name__ == '__main__':
    create_index()
    bulk_import_articles('crawled_data/articles')
    update_embeddings()
